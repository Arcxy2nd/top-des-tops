```javascript
function updateRowAvatar(sel, img) {
  const player = cachedPlayers.find(p => p.name === sel.value);
  const url = getAvatarUrl(sel.value, player ? player.meta : '');
  img.src = url;
  img.onerror = () => { img.src = getAvatarUrl(sel.value, ''); };
  // also update row background
  const row = sel.closest('.entry-row');
  if (row) {
    const bg = row.querySelector('.row-avatar-bg');
    if (bg) bg.style.backgroundImage = 'url(' + url + ')';
  }
}
```

### Task 3: Fix Chart.js hover — Ranking horizontal bars

**Files:** Modify `Index.html` JS ~line 4214–4227

**Problem:** `baseOpts` sets `tooltip: { mode: 'index', intersect: false }` — on a horizontal bar chart (`indexAxis: 'y'`), Chart.js needs `axis: 'y'` to correctly track rows by Y position instead of X.

- [ ] Add `interaction` override to the ranking chart options
- [ ] Override `tooltip` for ranking to add `axis: 'y'`

```javascript
currentChart = new Chart(document.getElementById('mainChart').getContext('2d'), {
  type: 'bar',
  data: { labels: totals.map(t => t.player), datasets },
  options: {
    ...baseOpts,
    indexAxis: 'y',
    interaction: { mode: 'index', axis: 'y', intersect: false },
    plugins: {
      ...baseOpts.plugins,
      tooltip: { mode: 'index', axis: 'y', intersect: false },
      legend: { display: chartOptions.detailed, labels: { color: c.text } }
    },
    scales: {
      x: { stacked: chartOptions.detailed, grid: { color: c.grid }, ticks: { color: c.muted } },
      y: { stacked: chartOptions.detailed, grid: { color: c.grid }, ticks: { color: c.muted } }
    }
```
